    一、SystemC专有数据类型

    • sc_bit：2值单比特（1、0）
    • sc_logic：4值单比特（1、0、X、Z）
    • sc_int：1到64比特有符号整型（即类似于C++中int通常对8/16/32位的数据操作）
    • sc_uint：1到64比特无符号整型
    • sc_bigint：任意宽度的有符号整型（不限制位数）
    • sc_biuint：任意宽度的无符号整型
    //例1
sc_int<4> x, y;
sc_int<8> z;
z = (x,y);       //将x和y转脸后赋值给z
x=z.range(7,4);  //取z的高4位赋值给x
    • sc_bv：任意宽度的2值比特向量（适用于只需要二值逻辑向量而不需要算数运算的场合）
    • sc_lv：任意宽度的4值比特向量
    //例2
sc_bv<6> val;
val="110011";
bool temp=val.xor_reduce();  //将val的各个位异或后赋值给temp
    • sc_fixed：模板类有符定点
    • sc_ufixed：模板类无符号定点
    • sc_fix：非模板类有符定点
    • sx_ufix：非模板类无符号定点

    二、进程（process）

    进程是程序在并发环境中的执行过程，具有动态性、并发性、独立性、异步性、结构性。

    在SystemC中，进程不是层次化的，即一个进程中不能包含或直接调用其它进程，但进程可以调用非进程的函数和方法。

    • 进程：
      1、动态概念；
      2、强调执行，是程序在数据集上的一次执行，有创建、有撤销，存在是暂时的；
      3、有并发性；
      4、竞争计算机资源的基本单位；
      5、进程是程序在数据集上的一次执行；
      程序：
      1、静态概念；
      2、指令的有序集合，永远存在；
      3、无并发性；
      4、不是；
      5、一个程序可对应多个进程，一个进程可包括多个程序；

    2.1 方法进程：SC_METHOD

    特点：唯一的可以综合的REL级别进程，当敏感列表中的事件发生时，它会被调用，调用后会立刻返回。该进程中不能使用 wait() 类语句。

    //用DC_METHOD进程构建异步D触发器
SC_MODULE(dff){}
        sc_in_clk clk;
        sc_in<bool> rst;
        sc_in<bool> din;
        sc_in<bool> dout;

        void do_it(){
                    if (rst)
                            dout = false;
                    else
                            dout = din;
        };

        SC_CTOR(dff){
                SC_METHOD(do_it);  //do_it时SC_METHOD进程，当clk或rst有上升沿发生时，do_it就会被触发执行
                sensitive_pos(rst);
                sensitive_pos(clk);
    }
};

    2.2 线程进程：SC_THREAD

    特点：能够被挂起和重新激活。

    SC_THREAD 使用 wait() 挂起，当敏感列表中有事件发生，线程进程被重新激活运行，当遇到新的 wait() 语句再重新挂起。

    多用于 Testbench 中输入激励和输出获取。

    //用DC_THREAD进程构建一个监视全加器输入输出变化值的变化情况
//File1：monitor.h
#include"systemc.h"
SC_MODULE(Monitor){
        sc_in<sc_bit> m_a, m_b, m_cin, m_sum, m_cout;

        void prc_monitor();

        SC_CTOR(Monitor){
                SC_THREAD(prc_monitor); //进程对以下数据敏感，初始化时被执行，并挂起在wait()语句
                sensitive<<m_a, m_b, m_cin, m_sum, m_cout;
        }
};

//File2:monitor.cpp
#include"monitor.h"
void Monitor::prc_monitor(){
            while(1){
                cout<<"At time"<<sc_time_stamp()<<"::";
                cout<<"(a, b, carry_in):";
                cout<<m_a.read()<<m_b.read()<<m_cin.read()<<",";
                cout<<"(sum, carry_out):";
                cout<<m_sum.read()<<m_cout.read()<<endl;
                wait();
            }
}

    2.3 钟控线程进程：SC_CTHREAD

    clock thread process 特点：只能在时钟沿降时被触发或激活，这种行为更接近硬件。

    引入的目的：产生更好的行为综合（行为综合时从高层次描述综合为 RTL 级描述的过程）。

    //例子
SC_MODULE(Driver){
            void prc_diver();

            SC_CTOR(DRIVER){
                SC_CTHREAD(prc_driver, clk.ps()); //只有在时钟上升沿时prc_driver才被激活。
                //必须在指定线程名字的同时，指定时钟和它的边沿。
                //下降沿表示方法为 clk.neg()
            }
}

    2.4 wait_until()、wait()和 next_trigger()

    wait_until：将进程挂起直达指定的表达式的值为真，只能用于线程进程和钟控线程进程。

    //wait_until举例
wait_until(clock.delayed()==true && reset.de;ayed()==false);

    wait()：等待函数，只能用于线程进程和钟控线程进程。

    //1、等;待敏感列表中有事件发生；
wait()

//2、等待事件发生；
sc_event e1;
......
wait(e1);

//3、等待事件之一发生；
sc_event e1, e2, e3;
......
wait(e1|e1|e2);

//4、等待事件全部发生；
sc_event e1, e2, e3;
......
wait(e1 & e1 & e2);

//5、等待100ns后，后面的语句会被执行；
wait(100, SC_NS);

//6、如果100ns内事件e1发生，或时间超过了100ns，进程将被激活；
sc_event e1;
......
wait(100, SC_NS, e1); //e1、e2、e3也可以换成第3、4条所述

//7、sc_time定义等效替代
sc_time t(100, SC_NS);
wait(t, e1); //等效于wait(100, SC_NS, e1);

    next_trigger()：和 wait() 参数一样，只能用于SC_METHOD进程。

    特点：调用后立刻返回，在一个 SC_METHOD 进程中可以多次调用 next_trigger()，但由最后一个 next_trigger() 决定进程的下一次激活时间。

    三、波形追踪

    3.1 顶层函数sc_main()

    所有 SystemC 的最顶层函数都是 sc_main()，没有该函数的代码无法编译成功。

    作用：将设计中的所有模块连接在一起，并引入时钟和波形跟踪（wave trace）

    格式：int sc_main(int argc, char* argv[]) {…/Body}，前一个表示参数的数目，后一个是用字符串表示的参数的值。

    3.2 波形追踪

    #include <systemc.h>

int sc_main(int argc, char* argv[]) {
    // 创建VCD波形跟踪文件
    sc_trace_file *wf = sc_create_vcd_trace_file("counter");

    // 定义信号或模块
    sc_signal<bool> clk;
    sc_signal<int> counter;

    // 将信号与波形跟踪文件相关联，所有于wf关联的信号值都会被记录在counter.vcd文件中
    sc_trace(wf, clk, "clk");
    sc_trace(wf, counter, "counter");

    // 创建模块等其他初始化操作
    // ...

        // 开始仿真运行，持续1000个时间单位
    sc_start(1000, SC_NS);

    // 仿真结束后关闭波形跟踪文件
    sc_close_vcd_trace_file(wf);

    return 0; //返回之前必须关闭波形文件
}
